Router

Router Configuration

The routing is based on the react-router-dom. You can find all routes inside this file src/router/index.ts . Uein routing system uses code splitting for improved performance.

router.ts

import { createRouter, createWebHistory } from "vue-router";
import NProgress from "nprogress";

// ROUTE GROUP FILES
import { dashboardRoutes } from "./dashboardRoutes";
import { userRoutes } from "./userRoutes";
import { productRoutes } from "./productRoutes";
import { orderRoutes } from "./orderRoutes";
import { invoiceRoutes } from "./invoiceRoutes";
import { ecommerceRoutes } from "./ecommerceRoutes";
import { sessionRoutes } from "./sessionRoutes";
import { settingsRoutes } from "./settingsRoutes";
import { supportRoutes } from "./supportRoutes";

// ===================
declare module "vue-router" {
  interface RouteMeta {
    title?: string;
  }
}
// ===================

const router = createRouter({
  history: createWebHistory(),
  routes: [
    {
      path: "/",
      name: "Home",
      component: () => import("@/pages/home/index.vue"),
      meta: { title: "Uein - Admin & Client Dashboard Template" },
    },
    {
      path: "/dashboard",
      name: "Dashboards",
      redirect: "/learning-management",
      component: () => import("@/layout/DefaultLayout.vue"),
      children: [
        ...dashboardRoutes,
        ...userRoutes,
        ...productRoutes,
        ...orderRoutes,
        ...invoiceRoutes,
        ...settingsRoutes,
        ...ecommerceRoutes,
        ...supportRoutes,
      ],
    },

    ...sessionRoutes,

    {
      path: "/:pathMatch(.*)*",
      name: "NotFound",
      component: () => import("../pages/sessions/NotFound.vue"),
    },
  ],
});

router.beforeEach((to, _from, next) => {
  // START THE ROUTE PROGRESS BAR.
  if (to.name) NProgress.configure({ showSpinner: false }).start();

  // SET THE PAGE TITLE.
  if (to.meta.title) document.title = to.meta.title;

  next();
});

router.afterEach(() => {
  // COMPLETE THE ANIMATION OF THE ROUTE PROGRESS BAR.
  NProgress.done();
});

export default router;

How to add a new route?

To add a new route in your Vue Router configuration, you can follow the example below. Here is a step-by-step guide for adding a new route:

  • Open file src/router/index.ts and update the following file if you want to add more routes.
  • Add the new route within the routes array.

router.ts

import { createRouter, createWebHistory } from "vue-router";

// ROUTE GROUP FILES
const router = createRouter({
  history: createWebHistory(),
  routes: [
    {
      path: "/",
      name: "home",
      redirect: "/learning-management",
      component: () => import("@/layout/DefaultLayout.vue"),
    },
    // Existing routes...

    // Added new route
    {
      name: "new-route",
      path: "/new-route",
      component: () => import("@/pages/NewFile.vue"),
    },
  ],
});

export default router;
