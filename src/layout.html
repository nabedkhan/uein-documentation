Default Layout

Understand folder structure of template and what everything contains 

project
└─ src
   └─ layout
      ├─ header
      |  ├─ popovers
      |  ├─ Header.vue
      ├─ sidebar
      |  ├─ NavAccordion.vue
      |  ├─ NavItem.vue
      |  ├─ SidebarNavigation.vue
      ├─ DefaultLayout.vue
      ├─ Mobilesidebar.vue

<script setup lang="ts">
import Header from "./header/Header.vue";

      <RouterView />
    </div>
  </main>
</template>
